## 4. 组件设计

### 4.1 背光服务模块

#### 4.1.1 接口定义 (`application/service/backlight_service.h`)

```c
#pragma once

#include <stdint.h>
#include <stdbool.h>

/* 配置项 */
#ifndef BACKLIGHT_IDLE_TIMEOUT_MS
#define BACKLIGHT_IDLE_TIMEOUT_MS   5000   // 5秒无操作熄灭
#endif

#ifndef LOW_BATTERY_THRESHOLD
#define LOW_BATTERY_THRESHOLD       20     // 低电量阈值 (%)
#endif

/* 配置结构体 */
typedef struct {
    uint16_t idle_timeout_ms;      // 休眠超时 (ms)
    uint8_t  low_battery_threshold; // 低电量阈值 (%)
} backlight_service_config_t;

/* 初始化 */
void backlight_service_init(const backlight_service_config_t* config);

/* 主循环任务 */
void backlight_service_task(void);

/* 活动通知（按键/触控时调用） */
void backlight_service_note_activity(void);

/* 低电量模式控制 */
void backlight_service_set_low_battery(bool low);
bool backlight_service_is_low_battery(void);

/* 启用/禁用休眠功能 */
void backlight_service_set_enabled(bool enabled);
bool backlight_service_is_enabled(void);
```

#### 4.1.2 实现要点 (`application/service/backlight_service.c`)

```c
#include "backlight_service.h"
#include "backlight.h"
#include "timer.h"

static backlight_service_config_t config = {
    .idle_timeout_ms = BACKLIGHT_IDLE_TIMEOUT_MS,
    .low_battery_threshold = LOW_BATTERY_THRESHOLD
};
static uint32_t last_activity_ms = 0;
static bool low_battery_mode = false;
static bool sleep_active = false;
bool g_initialized = false;

void backlight_service_init(const backlight_service_config_t* cfg) {
    if (cfg) config = *cfg;
    last_activity_ms = timer_read32();
    g_initialized = true;
}

void backlight_service_note_activity(void) {
    if (!g_initialized) return;

    last_activity_ms = timer_read32();

    // 如果背光关闭且不是低电量模式，重新点亮
    if (sleep_active && !low_battery_mode) {
        backlight_enable();
        sleep_active = false;
    }
}

void backlight_service_set_low_battery(bool low) {
    low_battery_mode = low;

    if (low) {
        // 低电量时强制关闭背光
        backlight_disable();
        sleep_active = true;
    }
}

void backlight_service_task(void) {
    if (!g_initialized || low_battery_mode || sleep_active) {
        return;
    }

    // 检查是否超时
    if (backlight_is_enabled() &&
        timer_elapsed32(last_activity_ms) >= config.idle_timeout_ms) {
        backlight_disable();
        sleep_active = true;
    }
}
```

#### 4.1.3 层级边界与包装函数 (`middleware/keyboard/keyboard.c`)

为保持层级边界清晰，在 keyboard.c 中提供背光活动通知的包装函数：

```c
#include "backlight_service.h"

/**
 * @brief 通知背光服务有活动发生
 *
 * 此函数供 Middleware 层（如 kb_fn_action.c）调用，
 * 避免直接依赖 Application 层的 backlight_service。
 * 在按键/触控事件发生时调用，以重置背光休眠计时器。
 */
void keyboard_note_backlight_activity(void) {
    backlight_service_note_activity();
}

/* 在 keyboard_task() 中调用 backlight_service_task() */
void keyboard_task(void) {
    matrix_task();
    debounce_task();
    action_task();
    combo_task();

    // 调用背光服务任务（检查休眠超时）
    backlight_service_task();
}
```

**层级依赖图（修正后）**:
```
Application (output_service.c)
    │
    └──▶ backlight_service.c (Application Service)
              │
              └──▶ backlight.c (Driver)

Middleware (kb_fn_action.c)
    │
    └──▶ keyboard.c (wrapper: keyboard_note_backlight_activity)
              │
              └──▶ backlight_service.c (通过包装函数，保持边界清晰)
```

### 4.3 Fn 组合键扩展

#### 4.3.0 键码定义补充 (`keyboards/kb904/keycodes_kb904.h`)

```c
/* KB904 专用键码扩展 */
#ifndef KC_RIGHT_SHIFT
#define KC_RIGHT_SHIFT      0xE5    // 右 Shift 键位
#endif

#ifndef KC_RIGHT_ENTER
#define KC_RIGHT_ENTER      0xE6    // 右 Enter 键位
#endif

#ifndef KC_RIGHT_GUI
#define KC_RIGHT_GUI        0xE7    // 右 Cmd/Win 键位
#endif

#ifndef KC_APPLICATION
#define KC_APPLICATION      0x65    // 应用程序键（自定义键位置）
#endif

/* Consumer 键码 - Siri 唤起 */
#ifndef M_SIRI
#define M_SIRI              0x00CF  // HCI Consumer Code for Voice Command
#endif

/* Fn 功能键 */
#define S_FN_KEY            0x00F0  // Fn 键内部标识（与现有定义保持一致）
```

#### 4.3.1 新增 Action 函数 (`middleware/keyboard/combo/kb_fn_action.c`)

> **架构约束**: Middleware 层不能直接调用 Application 层的 `backlight_service`。
> 解决方案：在 `keyboard.c` 中提供包装函数 `keyboard_note_backlight_activity()`，
> 由该函数调用 `backlight_service_note_activity()`，保持层级边界清晰。

```c
#include "backlight.h"
// #include "backlight_service.h"  // 禁止：Middleware 不能直接依赖 Application
#include "keyboard.h"              // 通过 keyboard.h 获取包装函数
#include "battery.h"
#include "indicator.h"
#include "indicator_config.h"

/**
 * @brief Fn+右Shift: 切换背光亮度（4档循环）
 */
uint8_t Backlight_Level_Up(uint16_t* add_keys) {
    backlight_level_step();                   // OFF→LOW→MEDIUM→HIGH→OFF
    keyboard_note_backlight_activity();       // 通过包装函数重置休眠计时器
    return 0;
}

/**
 * @brief Fn+右Enter: 切换背光颜色（13色循环）
 */
uint8_t Backlight_Color_Next(uint16_t* add_keys) {
    if (backlight_is_enabled()) {
        backlight_color_step();               // 只有背光亮时才切换
    }
    keyboard_note_backlight_activity();       // 通过包装函数重置休眠计时器
    return 0;
}

/**
 * @brief Fn+右Cmd: 电量检查（闪烁1-4次）
 */
uint8_t Battery_Check(uint16_t* add_keys) {
    uint8_t percentage = battery_get_percentage();
    uint8_t blinks;

    // 计算闪烁次数
    if (percentage >= 75)      blinks = 4;
    else if (percentage >= 50) blinks = 3;
    else if (percentage >= 25) blinks = 2;
    else                       blinks = 1;

    // 触发指示灯闪烁
    ind_effect_t effect = IND_BLINK_CUSTOM(200, 200, blinks);
    indicator_set(LED_BAT, &effect);

    return 0;
}

/**
 * @brief 自定义键: 唤起 Siri (Consumer 0xCF)
 */
uint8_t Siri_Invoke(uint16_t* add_keys) {
    add_keys[0] = M_SIRI;  // HCI_CONSUMER_CODE = 0xCF
    return 1;
}
```

#### 4.3.2 Combo 映射表更新 (`middleware/keyboard/combo/kb_combo_map.c`)

> **Combo 宏说明**:
> - `COMBO(combo_array, trigger_type, action)` - 标准组合键，按下即触发
> - `COMBO_LONG_TICKS(combo_array, ticks, trigger_type, action)` - 长按组合键
>   - `ticks`: 长按时间（单位：系统 tick，通常 1 tick = 10ms）
>   - `trigger_type`: `LONG_PRESS_START`（开始时触发）或 `LONG_PRESS_HOLD`（持续触发）
>
> **示例**: `COMBO_LONG_TICKS(..., 3000, ...)` = 3000 ticks = 30 秒长按（如需 3 秒，应使用 300）

```c
// 新增组合键枚举
enum combos {
    // ... 现有项 ...

    // KB904 新增
    FN_RIGHT_SHIFT_ID,    // Fn+右Shift: 亮度切换
    FN_RIGHT_ENTER_ID,    // Fn+右Enter: 颜色切换
    FN_RIGHT_CMD_ID,      // Fn+右Cmd: 电量检查
    CUSTOM_KEY_ID,        // 自定义键: Siri

    COMBOS_END,
};

// 组合键定义
const uint16_t PROGMEM FN_RIGHT_SHIFT_combo[] = { S_FN_KEY, KC_RIGHT_SHIFT, COMBO_END };
const uint16_t PROGMEM FN_RIGHT_ENTER_combo[] = { S_FN_KEY, KC_RIGHT_ENTER, COMBO_END };
const uint16_t PROGMEM FN_RIGHT_CMD_combo[]   = { S_FN_KEY, KC_RIGHT_GUI, COMBO_END };
const uint16_t PROGMEM CUSTOM_KEY_combo[]     = { KC_APPLICATION, COMBO_END };  // 自定义键位置

// 组合键数组
combo_t key_combos[] = {
    // ... 现有项 ...

    [FN_RIGHT_SHIFT_ID] = COMBO(FN_RIGHT_SHIFT_combo, PRESS_DOWN, Backlight_Level_Up),
    [FN_RIGHT_ENTER_ID] = COMBO(FN_RIGHT_ENTER_combo, PRESS_DOWN, Backlight_Color_Next),
    [FN_RIGHT_CMD_ID]   = COMBO(FN_RIGHT_CMD_combo,   PRESS_DOWN, Battery_Check),
    [CUSTOM_KEY_ID]     = COMBO(CUSTOM_KEY_combo,     PRESS_DOWN, Siri_Invoke),
};
```

### 4.4 指示灯配置

#### 4.4.1 LED 配置 (`keyboards/kb904/config_led.h`)

```c
#pragma once
#include "gpio.h"

/* KB904: 4 颗 LED */
#define IND_LED_COUNT   4

/* LED 索引定义 */
#define LED_CAPS        0   // 白灯：大写锁定
#define LED_BT          1   // 蓝灯：蓝牙状态
#define LED_POWER_RED   2   // 红灯：电源/低电量
#define LED_POWER_GRN   3   // 绿灯：充满电（只读，TP4056控制）

/* 兼容性别名 */

#define LED_CHARGE      LED_POWER_RED

    /* LED_BT */         {B22, true},   // 蓝灯：蓝牙状态
```

### 4.5 低电量联动

#### 4.5.1 OSAL 事件定义 (`application/service/output_service.h`)

```c
typedef enum {
    // ... 现有事件 ...
    OUTPUT_LOW_BATTERY_EVT    = 1 << 8,  /**< 低电量事件 */
    OUTPUT_BATTERY_NORMAL_EVT = 1 << 9,  /**< 电量恢复正常事件 */
} output_task_event_t;
```

#### 4.5.2 Battery 发送事件 (`drivers/power/battery.c`)

```c
extern uint8_t output_taskID;

void battery_task(void) {
    // ... 现有电压检测逻辑 ...

    uint8_t current_state = battery_get_power_state();
    static uint8_t last_state = BAT_POWER_NORMAL;

    // 检测状态变化
    if (current_state != last_state) {
        last_state = current_state;

        if (current_state == BAT_POWER_LOW) {
            // 发送低电量事件
            OSAL_SetEvent(output_taskID, OUTPUT_LOW_BATTERY_EVT);
        } else if (current_state == BAT_POWER_NORMAL &&
                   last_state == BAT_POWER_LOW) {
            // 电量恢复
            OSAL_SetEvent(output_taskID, OUTPUT_BATTERY_NORMAL_EVT);
        }
    }
}
```

#### 4.5.3 Output Service 处理事件 (`application/service/output_service.c`)

```c
uint16_t output_process_event(uint8_t task_id, uint16_t events) {
    // ... 现有事件处理 ...

    if (events & OUTPUT_LOW_BATTERY_EVT) {
        // 关闭背光
        backlight_service_set_low_battery(true);
        // 红灯慢闪
        indicator_set(LED_BAT, &IND_BLINK_SLOW);
        return (events ^ OUTPUT_LOW_BATTERY_EVT);
    }

    if (events & OUTPUT_BATTERY_NORMAL_EVT) {
        // 恢复背光
        backlight_service_set_low_battery(false);
        // 熄灭红灯
        indicator_set(LED_BAT, &IND_OFF);
        return (events ^ OUTPUT_BATTERY_NORMAL_EVT);
    }

    return 0;
}
```

### 5.3 错误处理

| 错误类型 | 检测方式 | 处理策略 |
|----------|----------|----------|
| EEPROM 写入失败 | CRC 校验失败 | 重试 3 次，失败后使用默认值 |
| 电量 ADC 异常 | 值超出范围 | 返回 50% 默认值 |
| 蓝牙配对超时 | 60s 计时器 | 停止广播，进入休眠 |
| 蓝牙回连超时 | 60s 计时器 | 进入休眠，唤醒后重试 |

### 5.4 恢复出厂设置

```c
void factory_reset_execute(void) {
    // 1. 指示灯反馈
    indicator_set(LED_CAPS, &IND_BLINK_3);
    indicator_set(LED_BT, &IND_BLINK_3);

    // 2. 清除蓝牙配对记录
    bt_driver_clear_bonding();

    // 3. 重置存储为默认值
    storage_factory_reset();

    // 4. 重置背光（白光 2 档）
    backlight_set_preset_level(BL_LEVEL_MEDIUM);
    backlight_set_preset_color(BL_COLOR_WHITE);

    // 5. 延迟后重启
    wait_ms(500);
    system_restart();
}
```
